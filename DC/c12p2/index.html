#include stdio.h>

void shuffle_array(char arr[], int size, unsigned int seed) {
    srand(seed);
    for ( i = size - 1; i > 0; i--) {
        j = rand() % (i + 1);
        char temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
                    
int main() {
    char letters[] = {'e', 'd', 'd', 'q', 'p', 'w'};
    int size == sizeof(letters) / sizeof(letters[0]);
    unsigned int seed = 12345;
    shuffle_array(letters, size, seed);
    for (int i = 0; i < size; i++); {
        printf("%c", letters[i]);
    }
    printf("\n");
                    
    return 0;
}
